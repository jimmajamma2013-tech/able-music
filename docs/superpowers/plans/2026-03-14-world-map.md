> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Artist World Map (temporal calendar section) into `able-v6.html` and `admin.html`, with state-chip integration and landing page demo state.

**Architecture:** All code is in a single HTML file per surface. The world map reads from `able_v3_profile.moments[]` (auto-populated from releases/shows, extended by standalone moments). The detail panel is a half-sheet using the existing panel pattern. The state chip gains a State 4 (near-future moment ≤ 7 days). Admin gets a "Your World" section with 5-step creation bottom-sheet.

**Tech Stack:** Vanilla JS, CSS custom properties, localStorage. No external libraries. Playwright MCP for verification.

      fan:       { label: 'Fan list',       cls: 'wm-panel__access--fan'       },

    // Build moment list from profile (auto-populate from releases + shows)
    function buildWorldMapMoments(profile) {
      const manual = (profile.moments || []).filter(m => m.active !== false)
      const today = isoDate(new Date())

      // Auto-create Release moment if not already linked
      const hasRelease = manual.some(m => m.linkedId === 'release')
      if (!hasRelease && profile.releaseDate) {
        manual.push({
          id: 'auto_release',
          type: 'release',
          date: profile.releaseDate,
          title: profile.releaseName || 'New release',
          access: 'public',
          cta: { label: profile.releaseDate > today ? 'Pre-save' : 'Stream now', url: profile.ctaUrl || '', type: 'link' },
          linkedId: 'release',
          isAuto: true,
        })
      }

      // Auto-create Show moments
      ;(profile.shows || []).forEach(show => {
        if (!manual.some(m => m.linkedId === show.id)) {
          manual.push({
            id: `auto_show_${show.id}`,
            type: 'show',
            date: show.date,
            title: show.venue || 'Live show',
            access: 'public',
            cta: { label: 'Get tickets', url: show.ticketUrl || '', type: 'link' },
            linkedId: show.id,
            isAuto: true,
          })
        }
      })

      return manual.sort((a, b) => a.date.localeCompare(b.date))
    }

    // Select featured moment for state-chip State 4
    function selectFeaturedMoment(moments) {
      const today = isoDate(new Date())
      const in7   = isoDate(new Date(Date.now() + 7 * 86400000))
      return moments.find(m =>
        m.active !== false &&
        m.date >= today &&
        m.date <= in7 &&
        m.access !== 'private'
      ) || null
    }

    // Resolve access gate for a moment
    function resolveAccess(moment, userState) {
      // userState: { isFan: bool, isSupporter: bool }
      switch (moment.access) {
        case 'public':    return 'full'
        case 'fan':       return userState.isFan ? 'full' : 'gate_fan'
        case 'supporter': return userState.isSupporter ? 'full' : 'gate_supporter'
        case 'private':   return 'enquire'
        default:          return 'full'
      }
    }

    // ISO date string from a Date object
    function isoDate(d) {
      return d.toISOString().slice(0, 10)
    }

    // Build month grid for given year/month, return { cells, hasMoments }
    // Returns Mon-first week (ISO week start)
    function buildMonthGrid(year, month, moments) {
      const today = isoDate(new Date())
      const firstDay = new Date(year, month, 1)
      // Monday = 0 ... Sunday = 6 offset
      let startOffset = firstDay.getDay() - 1
      if (startOffset < 0) startOffset = 6  // Sunday → last position

      const daysInMonth = new Date(year, month + 1, 0).getDate()
      const cells = []

      // Empty prefix cells
      for (let i = 0; i < startOffset; i++) cells.push(null)

      // Day cells
      for (let d = 1; d <= daysInMonth; d++) {
        const dateStr = `${year}-${String(month + 1).padStart(2,'0')}-${String(d).padStart(2,'0')}`
        const dayMoments = moments.filter(m => m.date === dateStr)
        cells.push({ day: d, dateStr, moments: dayMoments, isToday: dateStr === today })
      }

      // Pad to full rows of 7
      while (cells.length % 7 !== 0) cells.push(null)

      const hasMoments = moments.some(m => {
        const mDate = new Date(m.date)
        return mDate.getFullYear() === year && mDate.getMonth() === month
      })

      return { cells, hasMoments }
    }

    // Render grid cells into #wm-grid

```javascript
    // === YOUR WORLD (ADMIN) =====================================

    const WM_TYPE_COLOR_ADMIN = {
      release: '#3b82f6', show: '#e8715a', on_sale: '#f4b942',
      merch_drop: '#2db8b0', early_access: '#d4a84b',
      livestream: '#4b8fd4', private_access: '#888', announcement: '#aaa',
    }

    function renderYwMomentList(profile) {
      const list = document.getElementById('yw-moment-list')
      if (!list) return

      const moments = (profile.moments || []).filter(m => m.active !== false)
      const today = new Date().toISOString().slice(0,10)
      const upcoming = moments.filter(m => m.date >= today).sort((a,b) => a.date.localeCompare(b.date))

      if (!upcoming.length) {
        list.innerHTML = '<div class="yw-empty">No upcoming moments yet. Add your first.</div>'
        return
      }

      // Featured moment (within 7 days, non-private)
      const in7 = new Date(Date.now() + 7*86400000).toISOString().slice(0,10)
      const featuredId = upcoming.find(m => m.date <= in7 && m.access !== 'private')?.id

      list.innerHTML = upcoming.slice(0, 10).map(m => {
        const d = new Date(m.date + 'T00:00:00')
        const dateStr = d.toLocaleDateString('en-GB', { month: 'short', day: 'numeric' })
        const color = WM_TYPE_COLOR_ADMIN[m.type] || '#888'
        const isFeatured = m.id === featuredId
        return `<div class="yw-moment-row">
          <div class="yw-dot" style="background:${color}"></div>
          <div class="yw-moment-date">${dateStr}</div>
          <div class="yw-moment-title">${escA(m.title)}</div>
          <div class="yw-moment-access">${m.access}</div>
          ${isFeatured ? '<div class="yw-moment-featured">● In hero</div>' : ''}
        </div>`
      }).join('')

      // Private toggle state
      const toggle = document.getElementById('yw-private-toggle')
      if (toggle) toggle.checked = !!profile.privateAccessOpen

      // Enquiries
      const enquiries = (profile.privateAccessEnquiries || []).filter(e => e.status === 'pending')
      const enqEl = document.getElementById('yw-enquiry-count')
      if (enqEl) {
        if (enquiries.length) {
          enqEl.hidden = false
          enqEl.textContent = `${enquiries.length} enquir${enquiries.length === 1 ? 'y' : 'ies'} waiting`
        } else {
          enqEl.hidden = true
        }
      }
    }

    function initYourWorld(profile) {
      renderYwMomentList(profile)

      // Add moment button — simple prompt-based for v1
      document.getElementById('yw-add-btn')?.addEventListener('click', () => {
        // v1: simple prompt flow (full sheet in v2)
        const type = prompt('Moment type:\nrelease / show / on_sale / merch_drop / early_access / livestream / private_access / announcement', 'release')
        if (!type) return
        const date = prompt('Date (YYYY-MM-DD):', new Date().toISOString().slice(0,10))
        if (!date) return
        const title = prompt('Title (max 60 chars):', '')
        if (!title) return
        const access = prompt('Access:\npublic / fan / supporter / private', 'public')
        const url = prompt('CTA URL (optional):', '')
        const label = prompt('CTA label:', 'View')
        const p = safeLS(V3_KEY, {})
        const moments = p.moments || []
        moments.push({
          id: `m_${Date.now()}`,
          type: type.trim(),
          date: date.trim(),
          title: title.trim().slice(0,60),
          access: access?.trim() || 'public',
          cta: url ? { label: label || 'View', url: url.trim(), type: 'link' } : null,
          active: true,
        })
        p.moments = moments
        setLS(V3_KEY, p)
        renderYwMomentList(p)
      })

      // Private access toggle
      document.getElementById('yw-private-toggle')?.addEventListener('change', e => {
        const p = safeLS(V3_KEY, {})
        p.privateAccessOpen = e.target.checked
        setLS(V3_KEY, p)
      })
    }
```

Note: the `prompt()`-based creation is intentional for v1 — the 5-step bottom sheet is a Phase 2 admin upgrade. This ships the feature quickly without blocking on full admin UI.
